\section{Work in Progress}
\cvitem{Work in Progress}{Image Effects in Social Decisions}
\cvitem{}        {\textit{By investigating self-image concerns in the concept of strategic ignorance and by comparing the effects of ex-ante and ex-post ignorance the project aims to find answers to following questions:} \textit{Do people act pro-socially not to suffer from the action(letting others down or causing inequality) or do they compelled to do so in certain situations to protect their social /self image?} \textit{Do they avoid observing the consequences of their choices in a rich moral context like reciprocal decisions?}}

\cvitem{Work in Progress}{Self-Image Considerations in the Provision of Helpful Feedback - joint with Stefan Mei{\ss}ner}
